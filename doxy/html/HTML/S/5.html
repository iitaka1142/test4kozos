  60 /** get 1 thread from ready queue

  63  */
  64 static kz_thread * get( void )
  65 {
  66   kz_thread * result = readyque.head;
  67 
  68   if(readyque.head == NULL){
  69     // do not have to do anything
  70   }
  71   else if(readyque.head == readyque.tail){
  72     // queue has only 1 element and become empty
  73     readyque.head = readyque.tail = NULL;
  74   }
  75   else{
  76     readyque.head = readyque.head->next;

 159 static kz_thread_id_t thread_run(String name, kz_func_t func, int argc, char * argv[], stack_ptr_t stack_size)
 160 {
 161   kz_thread * thread = NULL;
 162   uint32 * sp = NULL;
 163   
 164   thread = get_vacant_TCB();
 165   if (thread == NULL) {
 166     /// did not find
 167     return -1;
 168   }
 169   
 170   // clear TCB
 171   memset(thread, 0, sizeof(*thread));
 172   
 173   // init thread parameters
 174   strcpy(thread->name, name);
 175   thread->init.func = func;
 176   thread->init.argc = argc;
 177   thread->init.argv = argv;
 178 
 179   // get stack area
 180   thread->stack = get_stack_area(stack_size);
 181   
 182   // initialize stack area
 183   sp = (uint32 *)thread->stack;
 184   (void)push(sp, (uint32)thread_end);
 185   
 186   // set program counter
 187   (void)push(sp, (uint32)thread_init);
 188   push(sp, 0); // ER6
 189   push(sp, 0); // ER5
 190   push(sp, 0); // ER4
 191   push(sp, 0); // ER3
 192   push(sp, 0); // ER2
 193   push(sp, 0); // ER1
 194 
 195   // arguments of thread_init)
 196   push(sp, (uint32)thread);
 197 
 198   // set context of thread
 199   thread->context.sp = (uint32)sp;
 200 
 201   // put thread to ready queue since that thread use system call
 202   put(current);
 203   
 204   // put new thread to ready queue
 205   put(thread);
 206 
 207   return (kz_thread_id_t)thread;
 208 }

 302 {
 303   // initialization
 304   current = NULL;
 305   readyque.head = readyque.tail = NULL;
 306   memset(threads, 0, sizeof(threads));
 307   memset(handlers, 0, sizeof(handlers));
