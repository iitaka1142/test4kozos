 115 int main(void)
 116 {
 117   static char buf[16];
 118   static long size = -1;
 119   static String loadbuf = NULL;
 120   char * entry_point;
 121   void (*f)(void);
 122         
 123   extern int buffer_start;
 124         
 125   INTR_DISABLE;
 126   init();
 127 
 128   /* 4th step */
 129   puts("kzload (kozos boot loader) started. \n");
 130 
 131   while (1) {
 132     puts("kzload> "); /* プロンプト表示 */
 133     gets(buf);
 134 
 135     if (!strcmp(buf, "load")) {
 136       /* ファイル転送 */
 137       loadbuf = (char *)(&buffer_start);
 138       size = xmodem_recv(loadbuf);
 139       wait();
 140       if (size < 0) {
 141         puts("\nXMODEM receive error!\n");
 142       } else {
 143         puts("\nXMODEM receive succeed.\n");
 144       }
 145     } else if (!strcmp(buf, "dump")) {
 146       /* バッファの16進ダンプ */
 147       puts("size : ");
 148       putxval(size, 0);
 149       puts("\n");
 150 
 151       dump(loadbuf, size);
 152     } else if (!strcmp(buf, "run")) {
 153       /* ELF形式ファイルをメモリ上に展開 */
 154       entry_point = elf_load(loadbuf);
 155       if(!entry_point) {
 156         puts("run error!");
 157       } else {
 158         puts("starting from entry_point ");
 159         putxval((uint32)entry_point, 0);
 160         puts("\n");
 161 
 162         f = (void (*)(void))entry_point;
 163         f(); /* ロードしたプログラムに処理を渡す */
 164       }
 165     } else if (!strcmp(buf, "exit")) {
 166       break;
 167     } else {
 168       puts("unknown.\n");
 169     }
 170   }
 171 
 172   return EXIT_SUCCESS;
 173 }
