87 static inline kz_thread * get( int priority )
88 {
89  if( priority < 0 || PRIORITY_NUM <= priority ) {
90  return -1;
91  }

94 
95  if(result == NULL){
96  // if queue is empty, return NULL
97  // (do not have to do anything)
98  }
99  else if(result->flag.ready == false){
100  // thread at top of queue hasn't been in ready state yet
101  }
102  else{
103  // set next item
104  readyque.head[priority] = readyque.head[priority]->next;
105 
106  // set ready flag as false
107  result->flag.ready = false;
108  }
109 
110  if(result->next == NULL){
111  // queue has only 1 element and become empty
112  readyque[priority].tail = NULL;
113  }
114 
115  return result;

179 {
180  kz_thread * thread = NULL;

182 
183  // get vacant TCB and clear it
184  thread = create_TCB();
185  if (thread == NULL) {
186  // did not find
187  return -1;
188  }
189 
190  // init thread parameters
191  strcpy(thread->name, name);
192  thread->next = NULL;
193  thread->priority = 0;
194  thread->flag.array = 0;
195  thread->init.func = func;
196  thread->init.argc = argc;
197  thread->init.argv = argv;
198 
199  // get stack area
200  thread->stack = get_stack_area(stack_size);
201 
202  // initialize stack area
203  sp = (uint32 *)thread->stack;
204  (void)push(sp, (uint32)thread_end);
205 
206  // set program counter
207  // in case of 0 priority, this thread isn't interrupted
208  // | CCR | PC | => | 0xc0 | &thread_init |
209  // 31 23 0 31 23 0
210  (void)push(sp, (uint32)thread_init | ((uint32)( (priority>0) ? 0 : 0xc0) << 24) );
211  (void)push(sp, 0); // ER6
212  (void)push(sp, 0); // ER5
213  (void)push(sp, 0); // ER4
214  (void)push(sp, 0); // ER3
215  (void)push(sp, 0); // ER2
216  (void)push(sp, 0); // ER1
217 
218  // arguments of thread_init (ER0)
219  (void)push(sp, (uint32)thread);
220 
221  // set context of thread
222  thread->context.sp = (uint32)sp;
223

225  put(current);
226

228  put(thread);
