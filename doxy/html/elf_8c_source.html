95  puts("magic is wrong\n");

97  puts("id class is wrong\n");
98  } else if (header->id.format != 2) {
99  puts("id format is wrong\n");
100  } else if (header->id.version != 1) {
101  puts("id version is wrong\n");
102  } else if (header->type != 2) {
103  puts("id type is wrong\n");
104  } else if (header->version != 1) {
105  puts("version is wrong\n");
106  } else if ((header->arch != 46) && (header->arch != 47)) {
107  puts("arch is wrong\n");
108  } else {
109  return EXIT_SUCCESS;
